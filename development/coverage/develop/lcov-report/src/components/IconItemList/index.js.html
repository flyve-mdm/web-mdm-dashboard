   * Convert arraybuffer to base64
   * @param {*} buffer
   * @return {string}
   */
  static arrayBufferToBase64(buffer) {
    let binary = ''
    const bytes = [].slice.call(new Uint8Array(buffer))
    bytes.forEach((b) => {
      binary += String.fromCharCode(b)
    })
    return window.btoa(binary)
  }
